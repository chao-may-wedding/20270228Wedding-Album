  <script>

    // ⚠️ 請在此處貼上您重新部署後取得的 Google Apps Script 網頁應用程式 URL

    const GAS_API_URL = 'https://script.google.com/macros/s/AKfycbzUOYVyFkIGrR4dGN9tiG5Vh6pGqddHd2UZxgCZGgYsCJl-G0IzqA70rhs2AOVvjiyYBg/exec'; 


    let uploadQueue = [];

    let totalFiles = 0;

    let uploadedCount = 0;


    const fileInput = document.getElementById('fileInput');

    const statusContainer = document.getElementById('statusContainer');

    const statusText = document.getElementById('status-text');

    const progressBar = document.getElementById('progressBar');


    fileInput.addEventListener('change', function(e) {

      const files = e.target.files;

      if (files.length === 0) return;


      totalFiles = files.length;

      uploadedCount = 0;

      uploadQueue = Array.from(files);

      

      statusContainer.style.display = 'block';

      updateProgressUI(0, '佇列準備中...');

      processUploadQueue();

    });


    function updateProgressUI(percent, message, color = '#f8fafc') {

      progressBar.style.width = percent + '%';

      statusText.innerText = message;

      statusText.style.color = color;

      

      if (color === '#ef4444') {

        progressBar.style.background = '#ef4444';

        progressBar.style.boxShadow = '0 0 10px #ef4444';

      } else if (color === '#10b981' || percent === 100) {

        progressBar.style.background = '#10b981';

        progressBar.style.boxShadow = '0 0 10px #10b981';

      } else {

        progressBar.style.background = 'var(--primary-color)';

        progressBar.style.boxShadow = '0 0 10px var(--primary-color)';

      }

    }


    // 使用標準 fetch 實作多檔案排隊上傳

    function processUploadQueue() {

      if (uploadQueue.length === 0) {

        updateProgressUI(100, `🎉 完美！${totalFiles} 個檔案已全數上傳完畢。`, '#10b981');

        fetchMedia(); 

        fileInput.value = ''; 

        setTimeout(() => { statusContainer.style.display = 'none'; }, 5000);

        return;

      }


      const file = uploadQueue.shift(); 

      const currentPercent = (uploadedCount / totalFiles) * 100;

      

      updateProgressUI(currentPercent, `⏳ 正在傳輸 (${uploadedCount + 1}/${totalFiles}):\n${file.name}`, '#38bdf8');


      const reader = new FileReader();

      reader.onload = function(event) {

        const base64Data = event.target.result;

        

        // 發送給 GAS doPost 的資料包

        const payload = {

          action: 'upload',

          base64Data: base64Data,

          fileName: file.name,

          mimeType: file.type

        };


        // 核心改變：使用 fetch 代替 google.script.run

        // ⚠️ 注意：不設 Content-Type 標頭以避開 CORS OPTIONS 預檢限制

        fetch(GAS_API_URL, {

          method: 'POST',

          body: JSON.stringify(payload)

        })

        .then(response => response.json())

        .then(res => {

          if(res.success) {

            uploadedCount++;

            const newPercent = (uploadedCount / totalFiles) * 100;

            updateProgressUI(newPercent, `✅ ${file.name} 傳送成功！\n冷卻中...`, 'var(--primary-color)');

            setTimeout(processUploadQueue, 3000); 

          } else {

            updateProgressUI(currentPercent, `⚠️ 伺服器忙碌 (${res.error})，重試中...`, '#ef4444');

            uploadQueue.unshift(file);

            setTimeout(processUploadQueue, 5000); 

          }

        })

        .catch(err => {

          updateProgressUI(currentPercent, `⚠️ 網路中斷或伺服器超載，重試中...`, '#ef4444');

          uploadQueue.unshift(file);

          setTimeout(processUploadQueue, 5000);

        });

      };

      reader.readAsDataURL(file);

    }


    // --- 輪播與手動切換邏輯 (改用 fetch 串接) ---

    let mediaFiles = [];

    let currentIndex = 0;

    let slideInterval;

    const SLIDE_DURATION = 6000;

    let touchStartX = 0;

    let touchEndX = 0;


    function fetchMedia() {

      const payload = { action: 'fetch' };


      fetch(GAS_API_URL, {

        method: 'POST',

        body: JSON.stringify(payload)

      })

      .then(response => response.json())

      .then(res => {

        if (res.success && res.data.length > 0) {

          mediaFiles = res.data;

          renderSlider();

        } else if (res.success) {

          document.getElementById('noMediaHint').innerHTML = '<span>尚無影像，快來上傳第一張照片吧！</span>';

        }

      })

      .catch(err => {

        console.error("無法載入雲端硬碟檔案列表:", err);

      });

    }


    function renderSlider() {

      const slider = document.getElementById('slider');

      const hint = document.getElementById('noMediaHint');

      if (hint) hint.style.display = 'none';


      const currentItemsCount = slider.querySelectorAll('.slide-item').length;

      if (currentItemsCount !== mediaFiles.length) {

        const items = slider.querySelectorAll('.slide-item');

        items.forEach(item => item.remove());

        

        mediaFiles.forEach((file, index) => {

          let element;

          if (file.mimeType.startsWith('image/')) {

            element = document.createElement('img');

            element.src = file.url;

          } else if (file.mimeType.startsWith('video/')) {

            element = document.createElement('video');

            element.src = file.url;

            element.controls = false;

            element.muted = true; 

            element.loop = true;

          }

          

          element.className = 'slide-item';

          if (index === currentIndex) element.classList.add('active');

          slider.appendChild(element);

        });

      }


      const prevBtn = document.getElementById('prevBtn');

      const nextBtn = document.getElementById('nextBtn');

      if (mediaFiles.length > 1) {

        prevBtn.classList.remove('hidden');

        nextBtn.classList.remove('hidden');

        resetTimer(); 

      } else {

        prevBtn.classList.add('hidden');

        nextBtn.classList.add('hidden');

      }

    }


    function changeSlide(nextIndex) {

      const items = document.querySelectorAll('.slide-item');

      if (items.length === 0) return;


      items[currentIndex].classList.remove('active');

      if (items[currentIndex].tagName === 'VIDEO') {

        items[currentIndex].pause();

      }


      currentIndex = nextIndex;

      items[currentIndex].classList.add('active');

      if (items[currentIndex].tagName === 'VIDEO') {

        items[currentIndex].play();

      }

      resetTimer();

    }


    function nextSlide() {

      if(mediaFiles.length === 0) return;

      let nextIndex = (currentIndex + 1) % mediaFiles.length;

      changeSlide(nextIndex);

    }


    function prevSlide() {

      if(mediaFiles.length === 0) return;

      let nextIndex = (currentIndex - 1 + mediaFiles.length) % mediaFiles.length;

      changeSlide(nextIndex);

    }


    function resetTimer() {

      clearInterval(slideInterval);

      if (mediaFiles.length > 1) {

        slideInterval = setInterval(nextSlide, SLIDE_DURATION);

      }

    }


    const sliderContainer = document.getElementById('slider');

    sliderContainer.addEventListener('touchstart', function(e) {

      touchStartX = e.changedTouches[0].screenX;

    }, false);


    sliderContainer.addEventListener('touchend', function(e) {

      touchEndX = e.changedTouches[0].screenX;

      handleSwipe();

    }, false);


    function handleSwipe() {

      const swipeThreshold = 50;

      if (touchStartX - touchEndX > swipeThreshold) nextSlide();

      if (touchEndX - touchStartX > swipeThreshold) prevSlide();

    }


    // 啟動系統

    fetchMedia();

    setInterval(fetchMedia, 15000); 

  </script>
